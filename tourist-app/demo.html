Method 2: Programmatic API (Advanced)

Custom Integration

Full control with JavaScript API:

<div id="booking-container"></div>

<script src="https://embed.vai.studio/button.js"></script>
<script>
  const button = VAI.createBookingButton({
    operatorId: 'c78f7f64-cab8-4f48-9427-de87e12ec2b9',
    templateId: 'db896f3d-da54-4304-bdf4-9a8499c2a7de',
    buttonText: 'Reserve Your Spot',
    themeColor: '#FF5722',
    onBookingComplete: (data) => {
      console.log('Booking completed!', data);
      alert('Thank you! Booking ID: ' + data.bookingId);
    }
  });

  document.getElementById('booking-container')
    .appendChild(button);
</script>
